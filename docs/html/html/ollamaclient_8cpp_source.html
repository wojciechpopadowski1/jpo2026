35void OllamaClient::generateCode(const QString &prompt, const QString &model)
36{
37 if (prompt.trimmed().isEmpty()) {
38 emit errorOccurred("Prompt nie może być pusty!");
39 return;
40 }
41
42 // Anuluj poprzednie żądanie jeśli istnieje
43 if (m_currentReply) {
44 m_currentReply->abort();
45 m_currentReply->deleteLater();
46 m_currentReply = nullptr;
47 }
48
49 QString modelToUse = model.isEmpty() ? Config::DEFAULT_MODEL : model;
50
51 emit progressUpdate("Połączenie z Ollama API...");
52 qDebug() << "OllamaClient: Wysyłanie promptu do modelu" << modelToUse;
53
54 // Przygotowanie żądania
55 QJsonObject requestJson = createRequestJson(prompt, modelToUse);
56 QJsonDocument doc(requestJson);
57 QByteArray jsonData = doc.toJson();
58
59 QNetworkRequest request;
60 request.setUrl(QUrl(Config::OLLAMA_API_URL));
61 request.setHeader(QNetworkRequest::ContentTypeHeader, "application/json");
62 request.setTransferTimeout(Config::NETWORK_TIMEOUT);
63
64 // Wysyłanie POST
65 m_currentReply = m_networkManager->post(request, jsonData);
66
67 emit progressUpdate("Oczekiwanie na odpowiedź z API...");
68}

80void OllamaClient::onReplyFinished(QNetworkReply *reply)
81{
82 if (reply != m_currentReply) {
83 reply->deleteLater();
84 return;
85 }
86
87 m_currentReply = nullptr;
88
89 // Obsługa błędów sieciowych
90 if (reply->error() != QNetworkReply::NoError) {
91 QString errorMsg;
92 if (reply->error() == QNetworkReply::ConnectionRefusedError) {
93 errorMsg = "Błąd: Nie można połączyć się z Ollama API. Upewnij się, że Ollama jest uruchomiona (http://localhost:11434).";
94 } else if (reply->error() == QNetworkReply::TimeoutError) {
95 errorMsg = "Błąd: Przekroczono limit czasu połączenia. Spróbuj ponownie.";
96 } else {
97 errorMsg = QString("Błąd sieci: %1").arg(reply->errorString());
98 }
99
100 qDebug() << "OllamaClient:" << errorMsg;
101 emit errorOccurred(errorMsg);
102 reply->deleteLater();
103 return;
104 }
105
106 // Odczytanie odpowiedzi
107 QByteArray responseData = reply->readAll();
108 reply->deleteLater();
109
110 qDebug() << "OllamaClient: Otrzymano odpowiedź (" << responseData.size() << "bajtów)";
111
112 // Parsowanie JSON
113 QJsonParseError parseError;
114 QJsonDocument doc = QJsonDocument::fromJson(responseData, &parseError);
115
116 if (parseError.error != QJsonParseError::NoError) {
117 QString errorMsg = QString("Błąd parsowania JSON: %1").arg(parseError.errorString());
118 qDebug() << "OllamaClient:" << errorMsg;
119 emit errorOccurred(errorMsg);
120 return;
121 }
122
123 QJsonObject jsonObj = doc.object();
124
125 // Wydobycie kodu z odpowiedzi
126 if (jsonObj.contains("response")) {
127 QString code = jsonObj["response"].toString().trimmed();
128
129 // Usunięcie ewentualnych znaczników markdown
130 if (code.startsWith("```python")) {
131 code = code.mid(9); // Usuń ```python
132 }
133 if (code.startsWith("```")) {
134 code = code.mid(3); // Usuń ```
135 }
136 if (code.endsWith("```")) {
137 code = code.left(code.length() - 3); // Usuń końcowe ```
138 }
139 code = code.trimmed();
140
141 if (code.isEmpty()) {
142 emit errorOccurred("Otrzymano pusty kod z API!");
143 return;
144 }
145
146 qDebug() << "OllamaClient: Wygenerowano kod (" << code.length() << "znaków)";
147 emit progressUpdate("Kod wygenerowany pomyślnie!");
148 emit codeGenerated(code);
149 } else {
150 QString errorMsg = "Brak pola 'response' w odpowiedzi API";
151 qDebug() << "OllamaClient:" << errorMsg;
152 emit errorOccurred(errorMsg);
153 }
154}
